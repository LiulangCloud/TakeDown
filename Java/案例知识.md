# 2.Random生成随机数

![](D:\TakeDown\Java\image\Random生成随机数.png)

```java
package com.huan.random;

import java.util.Random;//导包
import java.util.Scanner;

public class RandomTest2 {
    public static void main(String[] args) {
        Random r = new Random();//获取随机数对象
        Scanner sc = new Scanner(System.in);
        int luckNum = r.nextInt(100) + 1;//获取0~100之间的随机数

        while (true) {
            System.out.println("请输入您猜测的数字: ");
            int Num = sc.nextInt();
            if (Num == luckNum) {
                System.out.println("恭喜您，猜对了!");
                break;
            } else if (Num > luckNum) {
                System.out.println("很遗憾，猜大了!");
            } else if (Num < luckNum) {
                System.out.println("很遗憾，猜小了!");
            }
        }
    }
}
```

# 4.评委打分案例

**需求**某歌唱比赛，需要开发一个系统：可以录入6名评委的打分，录入完毕后立即输出平均分做为选手得分。

```java
package com.huan.define;

import java.util.Scanner;

public class ArrayDemo4 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        double[] scores = new double[6];
        double sum = 0;
        for (int i = 0; i < scores.length; i++) {
            System.out.println("请输入分数: ");
            scores[i] = sc.nextDouble();
            sum += scores[i];
            if (i == scores.length - 1) {
                sum = sum / scores.length;
            }
        }
        System.out.println("最终得分:" +  sum);
    }
}
```

# 6.数组反转

```java
package com.huan.define;

public class ArrayDemo6 {
    public static void main(String[] args) {
        //数组反转
        int[] arr = {10,20,30,40,50};
        for (int i = 0, j = arr.length - 1; i < j; i++ ,j--) {
            int temp = arr[j];
            arr[j] = arr[i];
            arr[i] = temp;
        }
        System.out.println(arr[0]);
        System.out.println(arr[1]);
        System.out.println(arr[2]);
        System.out.println(arr[3]);
        System.out.println(arr[4]);
    }
}
```



# 7.随机排名

```java
package com.huan.define;

import java.util.Random;
import java.util.Scanner;

public class ArrayDemo7 {
    public static void main(String[] args) {
        Scanner sc= new Scanner(System.in);
        int[] scores = new int[5];
        Random r = new Random();
        int num = r.nextInt(5);
        for (int i = 0; i < scores.length; i++) {
            System.out.println("请输入工号: ");
            scores[i] = sc.nextInt();
        }
        for (int j = 0; j < scores.length; j++) {
            int temp = scores[num];
            scores[num] = scores[j];
            scores[j] = temp;
        }
        System.out.println("--------------------");
        System.out.println(scores[0]);
        System.out.println(scores[1]);
        System.out.println(scores[2]);
        System.out.println(scores[3]);
        System.out.println(scores[4]);
    }
}
```
